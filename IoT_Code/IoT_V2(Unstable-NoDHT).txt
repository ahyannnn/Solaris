#include <WiFi.h>
#include <WiFiUdp.h>
#include <NTPClient.h>			//NTPClient by Fabrice
#include <SPI.h>
#include <SD.h>				//Adafruit BusIO by Adafruit, Adafruit Unified Sensor by Adafruit
#include <Ds1302.h>			//Time by Micahel Margolis, RTClib by Adafruit, Ds1302 by Rafa Couto
#include <TinyGPSPlus.h>		//TinyGPSPlus by Mikal Hart
#include <HardwareSerial.h>
#include <ModbusMaster.h>		//ModbusMaster by Doc Walker

// ------------------- Wi-Fi -------------------
const char* ssid     = "-----------";
const char* password = "---------------";

// ------------------- NTP -------------------
WiFiUDP ntpUDP;
NTPClient timeClient(ntpUDP, "pool.ntp.org", 8 * 3600, 60000);

// ------------------- RTC -------------------
#define DS1302_CLK 25
#define DS1302_DAT 26
#define DS1302_RST 27
Ds1302 rtc(DS1302_RST, DS1302_CLK, DS1302_DAT);

// ------------------- SD -------------------
#define SD_CS   4
#define SD_MOSI 23
#define SD_MISO 19
#define SD_SCK  18

bool sdAvailable = false;

// ------------------- GPS -------------------
TinyGPSPlus gps;
HardwareSerial gpsSerial(1);

// ------------------- RS485 -------------------
HardwareSerial rs485Serial(2);
ModbusMaster node;

#define MAX485_DE_RE 33

// ------------------- FUNCTIONS -------------------
void preTransmission() {
  digitalWrite(MAX485_DE_RE, HIGH);
}

void postTransmission() {
  digitalWrite(MAX485_DE_RE, LOW);
}

// ------------------- SETUP -------------------
void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("===== MULTI SENSOR LOGGER =====");

  // RTC
  rtc.init();
  Serial.println("✅ RTC Ready");

  // WiFi
  WiFi.begin(ssid, password);
  Serial.print("Connecting WiFi");

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  Serial.println();

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("✅ WiFi connected");
    timeClient.begin();

    if (timeClient.update()) {
      unsigned long epoch = timeClient.getEpochTime();

      Ds1302::DateTime dt;
      dt.second = epoch % 60;
      dt.minute = (epoch / 60) % 60;
      dt.hour   = (epoch / 3600) % 24;
      dt.day    = 30;
      dt.month  = 3;
      dt.year   = 26;
      dt.dow    = 1;

      rtc.setDateTime(&dt);
    }
  }

  // SD
  SPI.begin(SD_SCK, SD_MISO, SD_MOSI, SD_CS);
  if (SD.begin(SD_CS)) {
    sdAvailable = true;
    Serial.println("✅ SD OK");
  }

  // GPS
  gpsSerial.begin(9600, SERIAL_8N1, 16, 17);
  Serial.println("✅ GPS initialized");

  // RS485
  pinMode(MAX485_DE_RE, OUTPUT);
  digitalWrite(MAX485_DE_RE, LOW);

  rs485Serial.begin(9600, SERIAL_8N1, 34, 32);

  node.begin(1, rs485Serial); // slave ID = 1
  node.preTransmission(preTransmission);
  node.postTransmission(postTransmission);

  Serial.println("✅ RS485 ready");

  Serial.println("Setup done.\n");
}

// ------------------- LOOP -------------------
void loop() {

  // ---------------- TIME ----------------
  Ds1302::DateTime now;
  rtc.getDateTime(&now);

  char timestamp[32];
  snprintf(timestamp, sizeof(timestamp),
           "20%02d/%02d/%02d %02d:%02d:%02d",
           now.year, now.month, now.day,
           now.hour, now.minute, now.second);

  // ---------------- GPS READ ----------------
  while (gpsSerial.available()) {
    gps.encode(gpsSerial.read());
  }

  float latitude = gps.location.isValid() ? gps.location.lat() : 0.0;
  float longitude = gps.location.isValid() ? gps.location.lng() : 0.0;

  // ---------------- PYRANOMETER READ ----------------
  uint8_t result;
  float irradiance = -1;

  result = node.readHoldingRegisters(0x0000, 2); // adjust register!

  if (result == node.ku8MBSuccess) {
    uint16_t high = node.getResponseBuffer(0);
    uint16_t low  = node.getResponseBuffer(1);

    irradiance = ((high << 16) | low) / 10.0;
  }

  // ---------------- SERIAL DEBUG ----------------
  Serial.println("=================================");
  Serial.println(timestamp);

  Serial.print("📍 Lat: "); Serial.println(latitude, 6);
  Serial.print("📍 Lon: "); Serial.println(longitude, 6);

  if (irradiance >= 0) {
    Serial.print("☀️ Irradiance: ");
    Serial.print(irradiance);
    Serial.println(" W/m2");
  } else {
    Serial.println("⚠️ Pyranometer read failed");
  }

  // ---------------- LOGGING ----------------
  if (sdAvailable) {
    File file = SD.open("/log.txt", FILE_APPEND);

    if (file) {
      file.print(timestamp);
      file.print(",");

      file.print(latitude, 6);
      file.print(",");

      file.print(longitude, 6);
      file.print(",");

      file.println(irradiance);

      file.close();
      Serial.println("💾 Logged");
    } else {
      Serial.println("❌ SD write failed");
    }
  }

  delay(5000);
}
